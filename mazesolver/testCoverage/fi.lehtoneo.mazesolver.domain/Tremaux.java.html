package fi.lehtoneo.mazesolver.domain;

/**
 *
 * @author ossij
 */
public class Tremaux {
    char[][] maze;
    boolean[][] marked;
    boolean goingForward;
    public Tremaux(char[][] maze) {
        this.maze = maze;
        marked = new boolean[maze.length][maze[0].length];
        goingForward = true;
    }
    
    
    public String solve() {
        int[] current = new int[2];
        current[0] = 1;
        current[1] = 3;
        
        return "";
    }
    
    public boolean isIntersection(int x, int y) {
        if (maze[x - 1][y] == '.' || maze[x + 1][y] == '.' || maze[x][y-1] == '.' || maze [x][y + 1] == '.') {
            return true;
        }
        return false;
    }
    
    
    
    
}
